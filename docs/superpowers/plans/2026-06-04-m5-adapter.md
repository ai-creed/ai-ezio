# M5 — ai-whisper adapter Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make ai-ezio a first-class ai-whisper provider driven over the explicit JSONL protocol (no PTY/scraping), and prove it with one relay handoff end to end.

**Architecture:** A new `packages/adapter-ai-ezio` **in the ai-whisper repo** implements `CompanionProvider` + `InteractiveSessionController` from `@ai-whisper/shared`, backed by `@ai-ezio/harness`'s `Session`. Handoffs are delivered via `session.submit()`; turn completion comes from the explicit `idle` event (surfaced through a new optional `onTurnFinished` capability) instead of output quiescence. The ai-whisper mount runtime gains an additive `ai-ezio` drive path; the codex/claude byte-typing paths are untouched. `@ai-ezio/harness` stays workflow-agnostic; ai-ezio's placeholder `packages/adapter` is retired.

**Tech Stack:** TypeScript (NodeNext ESM), pnpm workspaces (both repos), vitest, zod (`mockProviderReplySchema`), `@ai-ezio/harness` + `@ai-ezio/protocol` (linked via pnpm `file:` overrides into ai-whisper).

**Source spec:** `docs/superpowers/specs/2026-06-04-m5-adapter-design.md`.

**Two repos, two histories.** ai-ezio changes commit in `/Users/vuphan/Dev/ai-ezio` (branch `master`); ai-whisper changes commit on a NEW branch in `/Users/vuphan/Dev/ai-whisper`. Never mix the two in one commit.

- **`ProviderReply`** = `{ kind, content (z.string().min(1)), transitionIntent (nullable) }`. Always build replies through `mockProviderReplySchema.parse(...)` so an invalid reply can never escape.

```ts
import { mockProviderReplySchema, type ProviderReply } from "@ai-whisper/shared";

const EMPTY_TURN_FALLBACK = "(no textual response; tool-only turn)";
const FAILURE_FALLBACK = "ai-ezio turn failed";

export type TurnOutcome =
	| { ok: true; content: string }
	| { ok: false; error: string };

/** Maps a finished ai-ezio turn to a schema-valid ProviderReply. A valid
 *  protocol event (including an empty tool-only turn) can never yield an
 *  invalid reply: content is always non-empty and kind is always a live
 *  replyKind. */
export function mapTurnToReply(outcome: TurnOutcome): ProviderReply {
	if (!outcome.ok) {
		return mockProviderReplySchema.parse({
			kind: "failure",
			content: outcome.error.length > 0 ? outcome.error : FAILURE_FALLBACK,
			transitionIntent: "failed",
		});
	}
	return mockProviderReplySchema.parse({
		kind: "answer",
		content: outcome.content.length > 0 ? outcome.content : EMPTY_TURN_FALLBACK,
		transitionIntent: null,
	});
}
```

```ts
import type { InteractiveSessionController } from "@ai-whisper/shared";
import type { ProtocolEvent } from "@ai-ezio/protocol";
import {
	defaultCreateEngineSession,
	type AiEzioEngineSession,
	type CreateEngineSession,
} from "./ai-ezio-engine.js";

export function createAiEzioLiveSession(input: {
	stdout: NodeJS.WritableStream;
	createEngineSession?: CreateEngineSession;
}): InteractiveSessionController {
	const create = input.createEngineSession ?? defaultCreateEngineSession;

	let session: AiEzioEngineSession | null = null;
	let lastContent = "";
	let sawTurn = false; // guards against the startup idle firing a handback

	const outputHandlers: Array<(data: string) => void> = [];
	const turnFinishedHandlers: Array<(content: string) => void> = [];
	const exitHandlers: Array<() => void> = [];

	const onEvent = (event: ProtocolEvent) => {
		switch (event.type) {
			case "assistant_delta": {
				sawTurn = true;
				for (const h of outputHandlers) h(event.text);
				input.stdout.write(event.text);
				break;
			}
			case "assistant_turn_finished": {
				sawTurn = true;
				lastContent = event.content;
				break;
			}
			case "idle": {
				if (!sawTurn) break; // startup idle — nothing to hand back
				const content = lastContent;
				lastContent = "";
				sawTurn = false;
				for (const h of turnFinishedHandlers) h(content);
				break;
			}
			default:
				break;
		}
	};

	return {
		async start() {
			session = create({ onEvent });
			session.onExit(() => {
				for (const h of exitHandlers) h();
			});
			await session.start();
		},
		async stop() {
			session?.close();
			session = null;
		},
		writeUserInput(data: string) {
			// Protocol-native: one submit, no keystream, no trailing CR.
			session?.submit(data);
		},
		sendLocalMessage(message: string) {
			// Local control text (relay preview) — surface to the operator's stdout.
			input.stdout.write(message);
		},
		onExit(handler: () => void) {
			exitHandlers.push(handler);
		},
		onProviderOutput(handler: (data: string) => void) {
			outputHandlers.push(handler);
		},
		onTurnFinished(handler: (content: string) => void) {
			turnFinishedHandlers.push(handler);
		},
	};
}
```

```js
import process from "node:process";
import { createAiEzioLiveSession } from "@ai-whisper/adapter-ai-ezio";
import { submitInjectedProviderInput } from "../packages/cli/src/runtime/provider-submit-strategy.js";

process.env.HAX_PROVIDER = "mock"; // deterministic engine, no LLM round-trip

const handbacks = [];
const live = createAiEzioLiveSession({ stdout: process.stdout });
live.onTurnFinished?.((content) => handbacks.push(content));

await live.start();

// Deliver a relay handoff exactly as the mount runtime does.
await submitInjectedProviderInput({
	target: "ai-ezio",
	text: "Say the word BANANA and nothing else.",
	writeUserInput: (t) => live.writeUserInput(t),
});

// Wait for the turn to finish (idle) — bounded.
const deadline = Date.now() + 15_000;
while (handbacks.length === 0 && Date.now() < deadline) {
	await new Promise((r) => setTimeout(r, 100));
}
await live.stop();

if (handbacks.length === 0) {
	console.error("FAIL: no handback captured from idle event");
	process.exit(1);
}
console.log("OK: relay handoff handed back via protocol:", JSON.stringify(handbacks[0]).slice(0, 120));
process.exit(0);
```
